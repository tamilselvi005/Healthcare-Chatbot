    public String respond(String message) {
        String msg = message.toLowerCase();

        if (checkGreeting(msg)) {
            return getRandom(greetings);
        } else if (checkFarewell(msg)) {
            return getRandom(farewells);
        } else if (checkThanks(msg)) {
            return getRandom(thanks);
        } else if (msg.contains("tablet")) {
            return "The " + tabletName + " tablet is used to treat various health conditions.";
        } else {
            return checkSymptoms(msg);
        }
    }

    private boolean checkGreeting(String msg) {
        String[] greetWords = {"hello", "hi", "hey", "howdy", "greetings"};
        return containsAny(msg, greetWords);
    }

    private boolean checkFarewell(String msg) {
        String[] byeWords = {"bye", "goodbye", "see you", "take care"};
        return containsAny(msg, byeWords);
    }

    private boolean checkThanks(String msg) {
        String[] thankWords = {"thank", "thanks", "thank you"};
        return containsAny(msg, thankWords);
    }

    private String checkSymptoms(String msg) {
        for (String symptom : symptoms.keySet()) {
            if (msg.contains(symptom)) {
                return symptoms.get(symptom);
            }
        }
        return getRandom(defaults);
    }

    private boolean containsAny(String msg, String[] words) {
        for (String w : words) {
            if (msg.contains(w)) return true;
        }
        return false;
    }

    private String getRandom(List<String> list) {
        Random r = new Random();
        return list.get(r.nextInt(list.size()));
    }

    public void chat() {
        Scanner sc = new Scanner(System.in);
        System.out.println(name + ": Hello! How can I help you today?");

        while (true) {
            System.out.print("You: ");
            String userInput = sc.nextLine();

            if (userInput.equalsIgnoreCase("exit")) {
                System.out.println(name + ": Goodbye!");
                break;
            }

            String response = respond(userInput);
            System.out.println(name + ": " + response);
        }

        sc.close();
    }
}
